dp_transfer_annex = {
	war_goal = annex_country
	texture = "gfx/interface/icons/war_goals/annex_country.dds"
	selectable_in_lens = {
		always = no
	}
	#possible = {
	#	always = yes
	#}
	on_weekly_pulse = {
		if = {
			limit = {
				escalation >= 60
				Not = {
					any_scope_play_involved = {
						NOT = {
							this = prev.target
						}
						is_diplomatic_play_enemy_of = prev.initiator
					}
				}
			}
			resolve_play_for = initiator
			end_play = yes
		}
		if = {
			limit = {
				any_scope_target_ally = {
					always = yes
				}
			}
			ordered_scope_target_ally = {
				order_by = prestige
				save_scope_as = new_leader
			}
			every_scope_target_ally = {
				limit = {
					NOT = {
						this = scope:new_leader
					}
				}
				root.initiator = {
					add_to_variable_list = {
						name = objectors
						target = prev
					}
				}
			}
			every_scope_initiator_ally = {
				root.initiator = {
					add_to_variable_list = {
						name = supporters
						target = prev
					}
				}
			}
			end_play = yes
			scope:new_leader = {
				create_diplomatic_play = {
					name = dp_transfer_liberate_subject_name
					target_country = root.target
					type = dp_transfer_liberate_subject
					add_initiator_backers = {
						root.target
					}
					add_war_goal = {
						holder = root.initiator
						type = annex_country
						target_country = root.target
					}
				}
			}
		}
	}
	on_war_begins = {

	}
	on_war_end = {

	}
}
